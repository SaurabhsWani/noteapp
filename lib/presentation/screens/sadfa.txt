BlocConsumer<AddnoteCubit, AddNoteState>(
              listener: (context, state) {
                final notificationSnackBar = SnackBar(
                  duration: Duration(milliseconds: 700),
                  content: Text(
                    'App ' + state.videoLinkcheck.toString().toUpperCase(),
                  ),
                );
                ScaffoldMessenger.of(context)
                    .showSnackBar(notificationSnackBar);
              },
              builder: (context, state) {
                if (state.videoLinkcheck == true) {
                  return Container(
                    child: Column(
                      children: [
                        SwitchListTile(
                          value: state.videoLinkcheck,
                          onChanged: (newValue) {
                            context.read<AddnoteCubit>().toggleVideo(newValue);
                          },
                          title: Text('App Notifications'),
                        ),
                      ],
                    ),
                  );
                }

                return Padding(
                  padding: const EdgeInsets.only(
                      left: 15.0, right: 15.0, top: 15, bottom: 0),
                  child: TextField(
                    onChanged: (value) {
                      titl = value;
                      setState(() {
                        titl = value;
                      });
                    },
                    keyboardType: TextInputType.text,
                    decoration: InputDecoration(
                        border: new OutlineInputBorder(
                          borderRadius: BorderRadius.all(Radius.circular(30)),
                          borderSide: BorderSide(
                            color: Colors.pinkAccent.shade400,
                          ),
                        ),
                        labelText: 'Video Link ',
                        prefixIcon: const Icon(
                          Icons.title,
                        ),
                        hintText: 'Paste Video Link Here'),
                  ),
                );
              },
            ),
